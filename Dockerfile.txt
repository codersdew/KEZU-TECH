# Use Node.js 20
FROM node:20

# Install ffmpeg for media processing
RUN apt-get update && apt-get install -y ffmpeg && rm -rf /var/lib/apt/lists/*

# Create app directory
WORKDIR /usr/src/app

# Install app dependencies
COPY package*.json ./
RUN npm install

# Bundle app source
COPY . .

# Expose port 5000 (Render uses PORT env var, but we default to 5000)
EXPOSE 5000

# Start the application
CMD [ "node", "index.js" ]
